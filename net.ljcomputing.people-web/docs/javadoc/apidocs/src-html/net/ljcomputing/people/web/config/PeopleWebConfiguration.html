017package net.ljcomputing.people.web.config;
018
019import javax.servlet.ServletContext;
020import javax.servlet.ServletContextEvent;
021import javax.servlet.ServletContextListener;
022import javax.servlet.ServletException;
023import javax.servlet.ServletRegistration;
024
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027import org.springframework.boot.SpringApplication;
028import org.springframework.boot.builder.SpringApplicationBuilder;
029import org.springframework.boot.context.web.SpringBootServletInitializer;
030import org.springframework.context.annotation.Bean;
031import org.springframework.context.annotation.ComponentScan;
032import org.springframework.context.annotation.Configuration;
033import org.springframework.context.annotation.Import;
034import org.springframework.web.WebApplicationInitializer;
035import org.springframework.web.context.support.AnnotationConfigWebApplicationContext;
036import org.springframework.web.servlet.DispatcherServlet;
037import org.springframework.web.servlet.config.annotation.EnableWebMvc;
038
039import net.ljcomputing.gson.config.GsonWebMvcConfigurerAdapter;
040import net.ljcomputing.people.PeopleApplication;
041
042/**
043 * People deployment configuration class.
044 * 
045 * @author James G. Willmore
046 *
047 */
048@Configuration
049@ComponentScan(basePackages = { "net.ljcomputing.core", "net.ljcomputing.people" })
050@EnableWebMvc
051@Import({ GsonWebMvcConfigurerAdapter.class, PeopleApplication.class })
052public class PeopleWebConfiguration extends SpringBootServletInitializer implements WebApplicationInitializer {
053        private static Logger logger = LoggerFactory.getLogger(PeopleWebConfiguration.class);
054
055        @Bean
056        protected ServletContextListener listener() {
057                return new ServletContextListener() {
058
059                        @Override
060                        public void contextInitialized(ServletContextEvent sce) {
061                                System.out.println("contextInitialized");
062                                logger.info("ServletContext initialized");
063                        }
064
065                        @Override
066                        public void contextDestroyed(ServletContextEvent sce) {
067                                System.out.println("contextDestroyed");
068                                logger.info("ServletContext destroyed");
069                        }
070
071                };
072        }
073
074        public static void main(String[] args) throws Exception {
075                SpringApplication.run(PeopleWebConfiguration.class, args);
076        }
077
078        protected SpringApplicationBuilder configure(SpringApplicationBuilder builder) {
079                System.out.println("configure");
080                logger.debug("Configuring ... ");
081                return builder.showBanner(true).sources(PeopleWebConfiguration.class).web(true).logStartupInfo(true);
082        }
083
084        public void onStartup(ServletContext container) throws ServletException {
085                logger.info("onStartup ...");
086                AnnotationConfigWebApplicationContext ctx = new AnnotationConfigWebApplicationContext();
087                ctx.register(PeopleWebConfiguration.class);
088                ctx.setServletContext(container);
089
090                DispatcherServlet ds = new DispatcherServlet(ctx);
091                ServletRegistration.Dynamic servlet = container.addServlet("dispatcherServlet", ds);
092
093                servlet.setInitParameter("throwExceptionIfNoHandlerFound", "true");
094                servlet.setLoadOnStartup(1);
095                servlet.addMapping("/");
096
097                logger.warn("servlet name: {}", servlet.getName());
098                logger.warn("servlet throwExceptionIfNoHandlerFound: {}",
099                                servlet.getInitParameter("throwExceptionIfNoHandlerFound"));
100        }
101
102}
